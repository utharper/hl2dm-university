net_graph "3" // [0,1,2,3] Net Graph.
net_graphpos "2" // [0,1,2,3] Set position on screen, 1 is bottom right, 2 is bottom middle and 3 is bottom left.
net_graphsolid "0" // [0,1] Net Graph Transparency.
net_graphheight "100" // [pixels] Vertical position of Net Graph.
net_scale "5" // [x] Set Net Graph vertical scale.

...put these in your autoexec.cfg, enjoy
