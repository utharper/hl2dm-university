anyone who's interested, here's my autoexec. Anything else not mentioned here is standard, except for a few graphics settings that are in my config

It's good for people running an outdated rig, for example myself. I have a [email protected], GeForce 6600GT, 1.5GB ram, 400Watt PS heap, but I've set it up so I can hold 60FPS (which is where my fps_max is set) constantly
I also have a few scripts, don't hate me for it...

Code: Select all

//Colin's custom setup
bind "SPACE" "+speed"
bind "mwheeldown" "+jump"
bind "mwheelup" "+jump"
bind "mouse3" "+use"
bind "x" "use weapon_smg1"
bind "c" "use weapon_ar2"
bind "e" "use weapon_rpg"
bind "r" "use weapon_frag"
bind "b" "use weapon_357"
bind "t" "+reload"
bind "q" "lastinv"
bind "f" "phys_swap"
bind "g" "impulse 100"
unbind "shift"
sensitivity "3"
cl_showfps 2
cl_showpos 1 
net_graph 0
cl_autowepswitch "0"
hud_quickinfo "0"		
hud_fastswitch 1
con_enable 1
bind "`" "toggleconsole"
name "Tranthor, the Space Hippo"
cl_playermodel "models/humans/Group03/Male_06.mdl"
fps_max 60
bind "F5" "jpeg"
jpeg_quality "100"

alias "+spinny" "cl_yawspeed 1500;  +left"
alias "-spinny" "-left;  cl_yawspeed 210"
bind "n" "+spinny"

alias "+rspinny" "cl_yawspeed 1500; +right"
alias "-rspinny" "-right; cl_yawspeed 210"
bind "m" "+rspinny"

alias "+spec" "spectate; spec0"
alias "-spec" "jointeam 0; spec1"
alias "spec0" "bind "k" "-spec""
alias "spec1" "bind "k" "+spec""
bind "k" "+spec"

alias "fix" "r_flushlod; snd_restart 1"
bind "l" "fix"


alias "zoomswitch1" "bind "z" "toggle_zoom"; bind "j" "zoomswitch0""
alias "zoomswitch0" "bind "z" "+zoom"; bind "j" "zoomswitch1""
bind "j" "zoomswitch1"

mat_clipz "1"
r_renderoverlayfragment "0"
r_decals "0"
mat_colorcorrection "0"
r_PhysPropStaticLighting "0"
r_3dsky "0"
r_maxmodeldecal "3"
r_eyemove "0"
r_eyeglintlodpixels "0"
r_waterforcereflectentities "0"
r_WaterDrawRefraction "0"
r_WaterDrawReflection "0"
cl_drawmonitors "0"
cl_ejectbrass "0"
muzzleflash_light "0"
violence_agibs "0"
violence_hgibs "0"
